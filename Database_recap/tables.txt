-------How to create Tables:---------------
Create Table employees (
    employee_id INT PRIMARY KEY, 
    first_name VARCHAR(10),    -- (10) means 10 is the max length first_name can have
    last_name VARCHAR(10), 
    hourly_date DECIMAL(5, 2), --(5,2) means the hourly_pay can have a total of 5 digits of which 2 will be after the decimal place.
    hire_date DATE, 
    email VARCHAR(50) UNIQUE
);

CONSTRAINTS: 
1. UNIQUE 
    i. Can be added when making a table e.g. 
        CREATE TABLE Employee(
            emp_id INT, 
            name VARCHAR(50) UNIQUE
        );
    ii. Can be added after making a table e.g. 
        ALTER TABLE Employee 
        ADD CONSTRAINT unique_name UNIQUE(name),
        ADD CONSTRAINT unique_email UNIQUE (email);

2. CHECK
    i. Can be added when making a table e.g.
        CREAT TABLE Employee(
            hourly_pay DECIMAL(5,2), 
            CHECK(hourly_pay >= 10.00)
            --you can also do CONSTRAINT chk_hourly_pay CHECK(hourly_pay >= 10.00) here
        );
    ii. Can be added after making a table e.g. 
        ALTER TABLE Employee 
        ADD CONSTRAINT chk_hourly_pay CHECK(hourly_pay >= 10.00);
    
3. PRIMARY KEY
    i. Can be added when making a table e.g.
        CREAT TABLE Employee(
            emp_id INT PRIMARY KEY
        );
    ii. Can be added after making a table e.g. 
        ALTER TABLE Employee 
        ADD CONSTRAINT pk_employee PRIMARY KEY(emp_id);

4. FOREIGN KEY 
    i. Can be added when making a table e.g.
        CREAT TABLE Employee(
            emp_id INT,
             FOREIGN KEY(emp_id) REFERENCES Workers(emp_id) ON DELETE SET NULL/CASCADE
        );
        -- On DElete set null --> sets the value of the foreign key to null if it is deleted from the original table
        -- on delete cascade deletes the row from the child table which has the deleted value as foreign key
    ii. Can be added after making a table e.g. 
        ALTER TABLE Employee 
        ADD CONSTRAINT fk_employee FOREIGN KEY(emp_id) REFERENCES Workers(emp_id) ON DELETE SET NULL/CASCADE;
    iii. ALTER TABLE DROP FOREIGN KEY fk_employee

3. NOT NULL
    i. Can be added when making a table e.g.
        CREAT TABLE Employee(
            emp_id INT NOT NULL
        );
    ii. Can be added after making a table e.g. 
        ALTER TABLE Employee 
        MODIFY emp_id INT NOT NULL,
        MODIY first_name VARCHAR(50) NOT NULL;

4. DEFAULT
    i. Can be added when making a table e.g.
        CREAT TABLE Employee(
            hourly_pay DECIMAL (5,2) DEFAULT 0
        );
    ii. Can be added after making a table e.g. 
        ALTER TABLE Employee 
        ALTER hourly_pay SET DEFAULT 0;


7. AUTO_INCREMENT/IDENTITY
    i. Can be added when making a table e.g.
        CREAT TABLE Employee(
            emp_id INT PRIMARY KEY AUTO_INCREMEMT; 
            -- or we can even do
            emp_od INT PRIMARY KEY AUTO_INCREMENT = 1000; --this will start the values from 1000
        );

But it would not be useful is you're just looking for a last_name

CREATE PROCEDURE employee_salary(
    IN emp INT,
    OUT pay DECIMAL(5,2)
)
BEGIN
    SELECT hourly_pay INTO pay
    FROM Employees
    WHERE emp_id = emp;
END $$

For example:

ORDER BY salary DESC
LIMIT 1 OFFSET 2
